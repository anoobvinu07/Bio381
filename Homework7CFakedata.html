n <- 30 # sample size
height <- runif(n, min = 0.2, max = 1.3) # height increment (dependent)
WUE <- runif(n, min = -0.01, max = 0.99) # wateruse efficiency (dependent)
temp <- runif(n, min = 4, max = 26) # temperature in degree celsius (independent)
temp1 <- rep(c("hot","cold"), 15)
ID <- seq_len(n)
expData <- data.frame(ID,temp,height,WUE)
